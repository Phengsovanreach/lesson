ផ្នែកទី ១៖ HTML (HyperText Markup Language)

HTML គឺ​ជា​គ្រឹះ​នៃ​គេហទំព័រ​ទាំង​អស់។ វា​ផ្តល់​នូវ​រចនាសម្ព័ន្ធ (Structure) សម្រាប់​មាតិកា​របស់​អ្នក។

១. រចនាសម្ព័ន្ធមូលដ្ឋាន និង <!DOCTYPE html>

ទំព័រ HTML ទាំងអស់ត្រូវតែមានរចនាសម្ព័ន្ធមូលដ្ឋាននេះ។ ការប្រកាស **<!DOCTYPE html>** ត្រូវតែជាបន្ទាត់ដំបូងបង្អស់ ហើយប្រាប់ Browser ថាឯកសារនេះជា HTML5 ។

<!DOCTYPE html>
<html lang="km">
<head>
    <meta charset="UTF-8">
    <title>ទំព័រទីមួយរបស់ខ្ញុំ</title>
</head>
<body>
    <!-- មាតិកាដែលអាចមើលឃើញនៅទីនេះ -->
</body>
</html>

ការមើលជាមុនផ្ទាល់៖

ទំព័រ HTML ត្រូវបានចាប់ផ្ដើម។ ចំណងជើងទំព័រ (Title) ត្រូវបានកំណត់ទៅជា "ទំព័រទីមួយរបស់ខ្ញុំ"។

២. ចំណងជើង (<h1> to <h6>) និង កថាខណ្ឌ (<p>)

**ចំណងជើង** ប្រើស្លាក `h1` ដល់ `h6` ដើម្បីបង្ហាញឋានានុក្រមនៃមាតិកា។ **h1** គឺសំខាន់បំផុត។ **កថាខណ្ឌ** ប្រើស្លាក `p` ។

<h1>នេះជាចំណងជើងសំខាន់បំផុត</h1>
<h3>ចំណងជើងរងកម្រិតទី ៣</h3>
<p>នេះ​គឺ​ជា​កថាខណ្ឌ​នៃ​អត្ថបទ​ដែល​មាន​ព័ត៌មាន​ពេញលេញ។</p>

៣. តំណភ្ជាប់ (<a>) និង រូបភាព (<img>)

ស្លាក **<a>** ប្រើគុណលក្ខណៈ **`href`** ដើម្បីបង្កើតតំណខ្ពស់។ ស្លាក **<img>** ប្រើ **`src`** សម្រាប់ប្រភពរូបភាព និង **`alt`** សម្រាប់អត្ថបទជំនួស (សំខាន់សម្រាប់ SEO) ។

<a href="https://www.google.com" target="_blank">ចូលទៅ Google</a>
<img src="https://placehold.co/100x50/10b981/ffffff?text=LOGO" alt="រូបភាពនិមិត្តសញ្ញា">

៤. បញ្ជី (<ul> <ol>)

**<ul>** សម្រាប់បញ្ជីដែលគ្មានលំដាប់ (ចំណុចមូល)។ **<ol>** សម្រាប់បញ្ជីដែលមានលំដាប់ (លេខ) ។ ធាតុនីមួយៗប្រើ **<li>** ។

<ul>
    <li>អង្ករ</li>
    <li>ត្រី</li>
</ul>
<ol>
    <li>ជំហានទី ១</li>
</ol>

៥. តារាងទិន្នន័យ (<table>)

តារាងប្រើ `table`, `tr` (ជួរដេក), `th` (ក្បាលជួរឈរ), និង `td` (ក្រឡាទិន្នន័យ) ។

<table>
    <tr>
        <th>ឈ្មោះ</th><td>ពិសិដ្ឋ</td>
    </tr>
</table>

៦. ទម្រង់ (<form>) និង ធាតុបញ្ចូលមូលដ្ឋាន

ស្លាក **<form>** ផ្ទុកធាតុបញ្ចូល។ **<label>** ផ្តល់ស្លាកឈ្មោះ។ **<input>** ប្រើ `type="text"` សម្រាប់អត្ថបទធម្មតា និង `type="email"` សម្រាប់អ៊ីមែល។

<form>
    <label for="name">ឈ្មោះ៖</label>
    <input type="text" id="name" required>
    <button type="submit">បញ្ជូន</button>
</form>

ការមើលជាមុនផ្ទាល់៖

៧. ធាតុបញ្ចូលកម្រិតខ្ពស់ (<radio>, <checkbox>, <select>)

ប្រើ `type="radio"` សម្រាប់ជ្រើសរើសជម្រើសតែមួយ និង `type="checkbox"` សម្រាប់ជ្រើសរើសច្រើន។ `<select>` ប្រើ `<option>` សម្រាប់បញ្ជីទម្លាក់ចុះ។

<input type="radio" name="g" value="male">ប្រុស
<input type="checkbox" checked>ទទួលព្រម
<select><option>ខ្មែរ</option></select>

៨. Semantic HTML (Header, Main, Footer)

ស្លាក Semantic ផ្តល់អត្ថន័យដល់រចនាសម្ព័ន្ធទំព័រ (មិនមែនគ្រាន់តែរចនាប័ទ្មទេ)។ **<header>** សម្រាប់ឡូហ្គោ, **<main>** សម្រាប់មាតិកាសំខាន់, **<footer>** សម្រាប់ព័ត៌មានរក្សាសិទ្ធិ។

<header>ឡូហ្គោគេហទំព័រ</header>
<main>មាតិកាសំខាន់នៅទីនេះ</main>
<footer>រក្សាសិទ្ធិ © ២០២៤</footer>

ការមើលជាមុនផ្ទាល់៖

Header: ឡូហ្គោ
Main: មាតិកាសំខាន់
Footer: រក្សាសិទ្ធិ

៩. Semantic HTML (Section, Article, Aside)

**<section>** សម្រាប់ដាក់ជាក្រុមមាតិកាដែលពាក់ព័ន្ធ។ **<article>** សម្រាប់មាតិកាឯករាជ្យ (ដូចជាអត្ថបទប្លុក)។ **<aside>** សម្រាប់ព័ត៌មានរឹម (Sidebar) ។

<section>
    <article>អត្ថបទពេញលេញ</article>
    <aside>ពាណិជ្ជកម្ម</aside>
</section>

ការមើលជាមុនផ្ទាល់៖

SECTION:
ARTICLE: រឿងរ៉ាវ
ASIDE: ព័ត៌មានបន្ថែម

១០. ធាតុពហុព័ត៌មាន (<audio> និង <video>)

ស្លាក `<audio>` និង `<video>` អនុញ្ញាតឱ្យអ្នកបង្កប់មេឌៀ។ គុណលក្ខណៈ **`controls`** ផ្តល់នូវរបារចាក់ (play/pause) សម្រាប់អ្នកប្រើប្រាស់។

<audio controls>
    <source src="song.mp3" type="audio/mpeg">
</audio>
<video width="200" controls>...</video>

ការមើលជាមុនផ្ទាល់៖

អ្នកនឹងឃើញការបញ្ជា (controls) សម្រាប់ចាក់មេឌៀដែលគ្មានប្រភព។

១១. Meta Tags និង Data Attributes (data-*)

**Meta tags** ដូចជា `viewport` (សម្រាប់ Responsive) គឺស្ថិតនៅក្នុង `<head>`។ **Data attributes** (`data-*`) អនុញ្ញាតឱ្យអ្នករក្សាទុកទិន្នន័យផ្ទាល់ខ្លួននៅលើធាតុ HTML ដែល JS អាចចូលប្រើបាន។

<!-- សម្រាប់ Mobile Responsive -->
<meta name="viewport" content="width=device-width, initial-scale=1.0">

<!-- Data Attribute -->
<div id="product" data-price="19.99">Item 1</div>

ការមើលជាមុនផ្ទាល់៖

ពិនិត្យមើលធាតុនេះសម្រាប់ data-id

ផ្នែកទី ២៖ CSS (Cascading Style Sheets)

CSS ទទួលខុសត្រូវក្នុងការកំណត់រចនាប័ទ្ម និងប្លង់ (Layout) នៃគេហទំព័ររបស់អ្នក។

១២. វិធីសាស្រ្តបញ្ចូល CSS (Inline, Internal, External)

មាន ៣ វិធីដើម្បីបន្ថែម CSS: **Inline** (នៅលើស្លាក), **Internal** (នៅក្នុងស្លាក `<style>`), **External** (ឯកសារ `.css` ដាច់ដោយឡែក)។

<!-- Internal CSS (ក្នុង <head>) -->
<style> .special { color: red; } </style>

<!-- Inline CSS -->
<p style="color: blue;">អត្ថបទពណ៌ខៀវ</p>

ការមើលជាមុនផ្ទាល់៖

អត្ថបទនេះប្រើ Inline CSS (ពណ៌ក្រហម)។

១៣. ពណ៌, ពុម្ពអក្សរ, និង Typography

អ្នកអាចផ្លាស់ប្តូរ **`color`** (អត្ថបទ), **`background-color`** (ផ្ទៃខាងក្រោយ), **`font-family`** (ប្រភេទពុម្ពអក្សរ) និង **`font-size`** ។

/* ឧទាហរណ៍ CSS */
.box {
    background-color: #34d399;
    color: white;
    font-family: Arial, sans-serif;
}

ការមើលជាមុនផ្ទាល់៖

នេះ​គឺ​ជា​អត្ថបទ​ដែល​មាន​ផ្ទៃ​ខាង​ក្រោយ​ពណ៌​បៃតង។

១៤. ឧបករណ៍ជ្រើសរើស CSS (Selectors)

Selectors កំណត់ថា CSS គួរអនុវត្តលើធាតុណា។ **Element Selector** (`p`), **Class Selector** (`.my-class`), **ID Selector** (`#my-id`) ។ Class ត្រូវបានប្រើញឹកញាប់បំផុត។

<style> #title { color: orange; } </style>
<h2 id="title">ចំណងជើងពិសេស</h2>

១៥. The Box Model (Margin, Border, Padding, Content)

រាល់ធាតុ HTML ទាំងអស់គឺជាប្រអប់ (Box) ដែលមាន ៤ ផ្នែក៖ **Content** (មាតិកា), **Padding** (ចន្លោះខាងក្នុង), **Border** (ស៊ុម), និង **Margin** (ចន្លោះខាងក្រៅ) ។

.box {
    padding: 15px; /* ចន្លោះខាងក្នុង */
    border: 2px solid black;
    margin: 20px; /* ចន្លោះខាងក្រៅ */
}

១៦. Display Property (Block, Inline, Inline-Block)

**`block`** យកទទឹងពេញមួយបន្ទាត់ (`div`, `p`) ។ **`inline`** យកតែទំហំមាតិកា (`span`, `a`) ។ **`inline-block`** យកទំហំមាតិកា តែអាចកំណត់ទទឹង/កម្ពស់បាន។

.item { display: inline-block; width: 100px; }

១៧. Flexbox៖ មូលដ្ឋានគ្រឹះ

Flexbox គឺជាឧបករណ៍ដ៏មានឥទ្ធិពលសម្រាប់ការតម្រៀបធាតុក្នុងមួយវិមាត្រ (ជួរដេក ឬជួរឈរ) ។ ចាប់ផ្តើមដោយកំណត់ **`display: flex`** លើធុងឪពុក (Parent) ។

.container { display: flex; flex-direction: row; }

១៨. Flexbox៖ ការតម្រឹម (Alignment & Justify)

**`justify-content`** គ្រប់គ្រងចន្លោះតាមអ័ក្សមេ (main axis) ។ **`align-items`** គ្រប់គ្រងការតម្រឹមតាមអ័ក្សឆ្លងកាត់ (cross axis) ។

.container {
    justify-content: center; /* ផ្ដេកកណ្តាល */
    align-items: center; /* បញ្ឈរកណ្តាល */
}

១៩. CSS Grid៖ សេចក្តីផ្តើម

Grid គឺជាឧបករណ៍សម្រាប់ប្លង់ពីរវិមាត្រ (ជួរដេក និងជួរឈរ) ។ កំណត់ **`display: grid`** និងប្រើ **`grid-template-columns`** ដើម្បីកំណត់ជួរឈរ។

.grid-cont {
    display: grid;
    grid-template-columns: 1fr 1fr 1fr; /* ៣ ជួរឈរស្មើគ្នា */

២០. Responsive Design: Media Queries

**Media Queries** អនុញ្ញាតឱ្យអ្នកអនុវត្តរចនាប័ទ្មផ្សេងៗដោយផ្អែកលើទំហំអេក្រង់ (Viewport) ។ នេះគឺចាំបាច់សម្រាប់ការមើលនៅលើទូរស័ព្ទ។

@media (max-width: 600px) {
    /* ស្ទីលនេះប្រើតែលើអេក្រង់តូចជាង ៦០០px ទេ */
    h1 { font-size: 1.5rem; }
}

២១. Transitions និង Animations

**`transition`** បង្កើតចលនាប្តូររលូននៅពេលស្ថានភាពផ្លាស់ប្តូរ (ឧ. លើ `hover`) ។ **`@keyframes`** អនុញ្ញាតឱ្យអ្នកបង្កើតចលនាដ៏ស្មុគស្មាញ។

.button {
    transition: background-color 0.3s ease; /* ភាពរលូន ០.៣ វិនាទី */
}
.button:hover { background-color: red; }

២២. Pseudo-classes (ឧ. :hover, :nth-child)

Pseudo-classes ជ្រើសរើសធាតុដោយផ្អែកលើស្ថានភាពរបស់វា។ **`:hover`** សម្រាប់ពេល Mouse ដាក់លើធាតុ។ **`:nth-child(n)`** ជ្រើសរើសធាតុរងតាមលំដាប់លំដោយ។

.list li:nth-child(odd) { background-color: #eee; } /* ជ្រើសរើសលេខសេស */
.link:focus { outline: 2px solid blue; } /* ពេលធាតុត្រូវបានជ្រើសរើស (Tabbed) */

ផ្នែកទី ៣៖ JavaScript (JS)

JavaScript គឺជាខួរក្បាលនៃគេហទំព័រ។ វាបន្ថែមអន្តរកម្ម និងអនុញ្ញាតឱ្យគេហទំព័រឆ្លើយតបទៅនឹងសកម្មភាពរបស់អ្នកប្រើប្រាស់។

២៣. សេចក្តីផ្តើម និងការបញ្ចូល JavaScript

JS អាចត្រូវបានបញ្ចូលតាមរយៈស្លាក **<script>** នៅក្នុង HTML ។ តាមទម្លាប់ ដាក់ស្លាកនេះនៅចុងបញ្ចប់នៃ `<body>` ដើម្បីឱ្យ HTML ផ្ទុកមុន។

<button onclick="console.log('Hello!')">Click</button>
<script>
    console.log('JS ដំណើរការ');
</script>

២៤. អថេរ (Variables) និង ប្រភេទទិន្នន័យ

ប្រើ **`let`** សម្រាប់អថេរដែលអាចផ្លាស់ប្តូរបាន និង **`const`** សម្រាប់តម្លៃដែលមិនផ្លាស់ប្តូរ (Constants) ។ ប្រភេទទិន្នន័យសំខាន់ៗ៖ **String** (អត្ថបទ), **Number** (លេខ), **Boolean** (`true`/`false`) ។

const appName = "My App";
let counter = 0;
let isActive = true;

២៥. ប្រតិបត្តិករ (Operators) និង លក្ខខណ្ឌ (Conditionals)

ប្រតិបត្តិករគណិតវិទ្យា (`+`, `-`, `*`) និងប្រៀបធៀប (`==`, `>`) ។ **`if / else`** អនុញ្ញាតឱ្យអ្នកប្រតិបត្តិប្លុកកូដដោយផ្អែកលើលក្ខខណ្ឌមួយ។

let age = 18;
if (age >= 18) {
    console.log("ពេញវ័យ");
} else {
    console.log("អនីតិជន");
}

២៦. អនុគមន៍ (Functions)

Functions គឺជាប្លុកកូដដែលអាចប្រើឡើងវិញបាន។ ពួកគេទទួល **Parameters** (ធាតុបញ្ចូល) និង **`return`** តម្លៃ (លទ្ធផល) ។

function greet(name) {
    return "ជំរាបសួរ, " + name;
}
let message = greet("មករា");

២៧. ព្រឹត្តិការណ៍ (Events)

Events គឺជារបៀបដែល JS ឆ្លើយតបទៅនឹងសកម្មភាពរបស់អ្នកប្រើប្រាស់។ ឧទាហរណ៍៖ **`click`** (ចុច), **`mouseover`** (ដាក់ Mouse លើ) ។

<button id="myBtn">...</button>

២៨. DOM Manipulation៖ ការជ្រើសរើសធាតុ

**DOM (Document Object Model)** គឺជាតំណាងធាតុ HTML របស់អ្នក។ អ្នកត្រូវតែជ្រើសរើសវាដើម្បីផ្លាស់ប្តូរវា។ វិធីទូទៅបំផុតគឺ **`document.getElementById()`** ។

const element = document.getElementById("myID");
const element2 = document.querySelector(".my-class"); /* ជ្រើសរើសដំបូងគេ */

២៩. DOM Manipulation៖ ការផ្លាស់ប្តូរមាតិកា

អ្នកអាចផ្លាស់ប្តូរអត្ថបទដោយប្រើ **`.textContent`** ឬ HTML ដោយប្រើ **`.innerHTML`** ។ អ្នកក៏អាចផ្លាស់ប្តូររចនាប័ទ្មដោយផ្ទាល់ដោយប្រើ **`.style`** ។

element.textContent = "New Text";
element.style.backgroundColor = "yellow";

៣០. DOM Manipulation៖ ការបង្កើតធាតុថ្មី

ដើម្បីបន្ថែមធាតុថ្មីទៅក្នុងទំព័រ៖ ប្រើ **`document.createElement()`** ដើម្បីបង្កើតស្លាក ហើយបន្ទាប់មក **`.appendChild()`** ដើម្បីភ្ជាប់វាទៅនឹងធាតុឪពុក។

const newP = document.createElement('p');
newP.textContent = "ធាតុថ្មី";
document.getElementById('parent').appendChild(newP);

៣១. Local Storage (ការរក្សាទុកទិន្នន័យក្នុង Browser)

**`localStorage`** អនុញ្ញាតឱ្យគេហទំព័ររបស់អ្នករក្សាទុកទិន្នន័យក្នុង Browser របស់អ្នកប្រើប្រាស់ ដែលនៅតែមានទោះបីជាបិទទំព័រហើយក៏ដោយ។ វាដំណើរការដោយការរក្សាទុក `key: value` ដែលជា String ។
